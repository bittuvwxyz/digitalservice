CREATE TABLE blogs (
    blog_id BIGINT PRIMARY KEY AUTO_INCREMENT,
    blog_title VARCHAR(255) NOT NULL,
    blog_short_description TEXT,
    blog_post LONGTEXT NOT NULL,
    blog_category VARCHAR(100),
    blog_author VARCHAR(100),
    blog_views BIGINT DEFAULT 0,
    blog_created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    blog_updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    ON UPDATE CURRENT_TIMESTAMP
);
